00151                         throw (ex.c_str());

00153                 else if(enn!=value.length()-1)
00154                 {
00155                         string ex = "invalid json - invalid literal found after string object '"+value+"' for name "+ element->getName();
00156                         throw (ex.c_str());
00157                 }
00158                 if(stn!=enn-1)
00159                 {
00160                         value = value.substr(stn+1, (enn-stn-1));
00161                 }
00162                 element->setType(JSONElement::JSON_STRING);
00163         }
00164         else if(StringUtil::toLowerCopy(value)=="true" || StringUtil::toLowerCopy(value)=="false")
00165         {

00169         else if(value.find(".")!=string::npos)
00170         {
00171                 try
00172                 {
00173                         CastUtil::lexical_cast<double>(value);

00175                         string exp = "invalid json - invalid double value "+value+" found for name "+ element->getName();
00176                         throw (exp.c_str());

00179         }
00180         else
00181         {
