33 bool Client::connection(string host,int port)
34 {
35  struct sockaddr_in *remote;
36  int tmpres;
37  char *ip;
38 
39  sockfd = create_tcp_socket();
40  ip = get_ip((char*)host.c_str());
41  fprintf(stderr, "IP is %s\n", ip);
42  remote = (struct sockaddr_in *)malloc(sizeof(struct sockaddr_in *));
43  remote->sin_family = AF_INET;
44  tmpres = inet_pton(AF_INET, ip, (void *)(&(remote->sin_addr.s_addr)));
45  if( tmpres < 0)
46  {
47  perror("Can't set remote->sin_addr.s_addr");
48  return false;
49  }
50  else if(tmpres == 0)
51  {
52  fprintf(stderr, "%s is not a valid IP address\n", ip);
53  return false;
54  }
55  remote->sin_port = htons(port);
56 
57  if(connect(sockfd, (struct sockaddr *)remote, sizeof(struct sockaddr)) < 0){
58  perror("Could not connect");
59  connected = false;
60  } else {
61  connected = true;
62  }
63  free(remote);
64  free(ip);
65 
66  return connected;
67 }

70 bool Client::connectionUnresolv(string host,int port)
71 {
72  struct addrinfo hints, *servinfo, *p;
73  int rv;
74  char s[INET6_ADDRSTRLEN];
75 
76  memset(&hints, 0, sizeof hints);
77  hints.ai_family = AF_UNSPEC;
78  hints.ai_socktype = SOCK_STREAM;
79  string sport = CastUtil::lexical_cast<string>(port);
80  if ((rv = getaddrinfo(host.c_str(), sport.c_str(), &hints, &servinfo)) != 0) {
81  fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(rv));
82  return false;
83  }
84 
85  // loop through all the results and connect to the first we can
86  for(p = servinfo; p != NULL; p = p->ai_next) {
87  if ((sockfd = socket(p->ai_family, p->ai_socktype,
88  p->ai_protocol)) == -1) {
89  perror("client: socket");
90  continue;
91  }
92 
93  if (connect(sockfd, p->ai_addr, p->ai_addrlen) == -1) {
94  close(sockfd);
95  perror("client: connect");
96  connected = false;
97  continue;
98  } else {
99  connected = true;
100  }
101  break;
102  }
103 
104  if (p == NULL) {
105  fprintf(stderr, "client: failed to connect\n");
106  return false;
107  }
108 
109  inet_ntop(p->ai_family, get_in_addr((struct sockaddr *)p->ai_addr),
110  s, sizeof s);
111  //printf("client: connecting to %s\n", s);
112 
113  freeaddrinfo(servinfo); // all done with this structure
114 
115  return connected;
116 }

142 string Client::getTextData(string hdrdelm,string cntlnhdr)
143 {
144  int er=-1;

147  int cntlen;
148  char buf[MAXBUFLE];
149  BIO* sbio=BIO_new_socket(sockfd,BIO_NOCLOSE);

151  BIO_push(io,sbio);
152  while(flag)
153  {
154  er = BIO_gets(io,buf,MAXBUFLE-1);
155  if(er==0)
156  {
157  if(io!=NULL)BIO_free_all(io);
158  //logger << "\nsocket closed before being serviced" <<flush;
159  return alldat;
160  }
161  if(!strcmp(buf,hdrdelm.c_str()))
162  {
163  string tt(buf, er);
164  alldat += tt;
165  break;
166  }
167  string temp(buf, er);
168  temp = temp.substr(0,temp.length()-1);
169  alldat += (temp + "\n");
170  if(temp.find(cntlnhdr)!=string::npos)
171  {
172  std::string cntle = temp.substr(temp.find(": ")+2);
173  cntle = cntle.substr(0,cntle.length()-1);
174  try
175  {
176  cntlen = CastUtil::lexical_cast<int>(cntle);
177  }
178  catch(...)
179  {
180  logger << "bad lexical cast" <<endl;
181  }
182  }
183  memset(&buf[0], 0, sizeof(buf));
184  }
185  while(cntlen>0)
186  {
187  //logger << "reading conetnt " << cntlen;
188  int toRead = cntlen;
189  if(cntlen>MAXBUFLE)
190  toRead = MAXBUFLE - 1;
191  er = BIO_read(io,buf,toRead);
192  if(er==0)
193  {
194  if(io!=NULL)BIO_free_all(io);
195  //logger << "\nsocket closed before being serviced" <<flush;
196  return alldat;
197  }
198  string temp(buf, er);
199  alldat += temp;
200  cntlen -= er;
201  memset(&buf[0], 0, sizeof(buf));
202  }
203  return alldat;
204 }
